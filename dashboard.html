def vec_running_avg(x, N_pts):
    N_pts_side = int(N_pts/2)
    x_m = np.empty(len(x)-N_pts)
    
    for i in range(len(x)-N_pts):
        tmpx = np.array(x[N_pts_side+i:N_pts_side+i+N_pts])
        x_m[i] = np.mean(tmpx)

    return x_m
